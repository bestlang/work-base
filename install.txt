拷贝bestlang和sniper目录
编辑composer.json的repositories
composer require bestlang/laracms
composer require sniper/employee
composer dump-autoload
修改.env
	-DB
	-PROXY_DB


拷贝resources/admin
npm install
npm run dev
修改./config/prod.env.js中SITE_URL
npm run build

发布资源
composer dump-autoload
php artisan vendor:publish --provider="BestLang\Base\Providers\AppServiceProvider" --force
php artisan vendor:publish --provider="BestLang\Laracms\Providers\AppServiceProvider"
php artisan vendor:publish --provider="BestLang\LaravelUEditor\UEditorServiceProvider"
php artisan vendor:publish --provider="BestLang\WxPay\WxPayServiceProvider"
php artisan vendor:publish --provider="Sniper\Employee\Providers\AppServiceProvider"

php artisan migrate
php artisan db:seed --class=BestLangBaseSeeder
php artisan db:seed --class=LaraCmsSeeder
php artisan db:seed --class=SniperSeeder
